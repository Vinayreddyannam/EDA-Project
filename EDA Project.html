Web Scraping and Exploratory Data Analysis of Smart TV Listings on Flipkart

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager
import pandas as pd
import time
import re

options = webdriver.ChromeOptions()
options.add_argument("--start-maximized")
options.add_argument("--disable-blink-features=AutomationControlled")

driver = webdriver.Chrome(
    service=Service(ChromeDriverManager().install()),
    options=options
)

wait = WebDriverWait(driver, 15)

# Lists for final dataset
tv_name_clean = []
brand = []
tv_size = []
price = []
ratings_count = []
reviews_count = []
highlights = []

TOTAL_PAGES = 10  

for page in range(1, TOTAL_PAGES + 1):

    print(f"Scraping page {page}...")
    url = f"https://www.flipkart.com/search?q=smart+tv&page={page}"
    driver.get(url)

    # Close login popup (only first time usually)
    try:
        close_btn = wait.until(
            EC.element_to_be_clickable((By.XPATH, "//button[contains(text(),'')]"))
        )
        close_btn.click()
    except:
        pass

    time.sleep(3)

    products = driver.find_elements(By.XPATH, "//div[@data-id]")
    print("Products found:", len(products))

    for product in products:

        # TV Name
        try:
            name = product.find_element(By.TAG_NAME, "a").text
            name = name.replace("Add to Compare", "").replace("\n", " ").strip()
        except:
            continue

        # Brand
        brand_name = name.split()[0]

        # TV Size (inch)
        size_match = re.search(r"\((\d+)\s*inch\)", name)
        size_inch = float(size_match.group(1)) if size_match else None

        # Price
        try:
            price_text = product.find_element(By.XPATH, ".//div[contains(text(),'')]").text
            price_value = float(price_text.replace("", "").replace(",", ""))
        except:
            price_value = None

        # Ratings & Reviews
        try:
            rr = product.find_element(By.XPATH, ".//span[contains(text(),'Ratings')]").text
            rr = rr.replace(",", "").split()
            r_count = float(rr[0])
            rev_count = float(rr[-2])
        except:
            r_count, rev_count = None, None

        # Highlights
        try:
            feats = product.find_elements(By.TAG_NAME, "li")
            highlight_text = " | ".join([f.text for f in feats if f.text])
        except:
            highlight_text = None

        # Append to lists
        tv_name_clean.append(name)
        brand.append(brand_name)
        tv_size.append(size_inch)
        price.append(price_value)
        ratings_count.append(r_count)
        reviews_count.append(rev_count)
        highlights.append(highlight_text)

driver.quit()

# Final DataFrame
final_df = pd.DataFrame({
    "TV_Name_Clean": tv_name_clean,
    "Brand": brand,
    "TV_Size_Inch": tv_size,
    "Price": price,
    "Ratings_Count": ratings_count,
    "Reviews_Count": reviews_count,
    "Highlights": highlights
})

print("\nTotal TVs scraped:", len(final_df))
final_df.to_csv("final_flipkart_tv_dataset.csv", index=False)
print("Saved as final_flipkart_tv_dataset.csv")
